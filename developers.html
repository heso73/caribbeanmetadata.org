03

Valider un fichier CMS

Python 
# pip install jsonschema requests
import requests, jsonschema, json

# 1. Charger le schéma CMS officiel
schema_url = "https://caribbeanmetadata.org/schema/v1.0/cms-metadata.json"
schema = requests.get(schema_url).json()

# 2. Charger votre fichier œuvre
with open("mon_oeuvre.json") as f:
    oeuvre = json.load(f)

# 3. Valider
try:
    jsonschema.validate(instance=oeuvre, schema=schema)
    print("✅ Fichier conforme au standard CMS v1.0")
except jsonschema.ValidationError as e:
    print(f"❌ Erreur : {e.message}")
