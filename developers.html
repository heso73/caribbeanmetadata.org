Exemples de code

Intégrer l'API

CURL — Œuvres par territoire
curl "https://api.caribbeanmetadata.org/api/v1/works?territory=Guadeloupe&limit=10"
CURL — Vérifier une certification
curl "https://api.caribbeanmetadata.org/api/v1/certifications/CMS-GUA-2026-XXXXXXXX"
CURL — Soumettre une œuvre (POST)
curl -X POST https://api.caribbeanmetadata.org/api/v1/works \
  -H "Content-Type: application/json" \
  -d '{
    "title": "Tambou a nou",
    "territory": "Guadeloupe",
    "languages": ["gcf"],
    "family": "music",
    "year": 2026
  }'
JAVASCRIPT — Fetch stats
// Récupérer les statistiques CMS en temps réel
const res = await fetch(
  'https://api.caribbeanmetadata.org/api/v1/stats'
);
const stats = await res.json();
console.log(`${stats.works.total} œuvres · ${stats.vocabulary.cultural_markers} marqueurs`);
JAVASCRIPT — Filtrer par territoire et langue
const params = new URLSearchParams({
  territory: 'Guadeloupe',
  language: 'gcf',
  limit: 20
});
const res = await fetch(
  `https://api.caribbeanmetadata.org/api/v1/works?${params}`
);
const { data } = await res.json();
data.forEach(work => console.log(work.title, work.territory));
JAVASCRIPT — Vérifier une certification
const cmsId = 'CMS-GUA-2026-XXXXXXXX';
const res = await fetch(
  `https://api.caribbeanmetadata.org/api/v1/certifications/${cmsId}`
);
const cert = await res.json();
if (cert.valid) {
  console.log(`✓ ${cert.work.title} — ${cert.work.territory}`);
}
PYTHON — Lister les marqueurs culturels
import requests

# Récupérer tous les marqueurs CMS officiels
r = requests.get(
  'https://api.caribbeanmetadata.org/api/v1/vocabularies/markers'
)
markers = r.json()['markers']
for m in markers:
    print(f"{m['id']} — {m['name']} ({m['category']})")
PYTHON — Valider un fichier CMS
import requests, jsonschema, json

# Charger le schéma CMS officiel
schema_url = "https://raw.githubusercontent.com/heso73/cms-nlp-corpus/main/data/CMS_corpus_schema_v2.json"
schema = requests.get(schema_url).json()

# Valider votre fichier œuvre
with open("mon_oeuvre.json") as f:
    oeuvre = json.load(f)

try:
    jsonschema.validate(instance=oeuvre, schema=schema)
    print("✅ Fichier conforme au standard CMS v2.0")
except jsonschema.ValidationError as e:
    print(f"❌ Erreur : {e.message}")
PYTHON — Filtrer le corpus par langue
import requests, pandas as pd

r = requests.get(
  'https://api.caribbeanmetadata.org/api/v1/works',
  params={'language': 'gcf', 'limit': 100}
)
works = r.json()['data']
df = pd.DataFrame(works)
print(df[['title', 'territory', 'family']])
